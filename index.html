let expr = <@@ if 10 < 5 then "really?!" else "good" @@>

let rec traverse expr =
  match expr with
  | ShapeVar (v) ->
    Expr.Var(v)
  | ShapeLambda(v, body) ->
    Expr.Lambda(v, traverse body)
  | ShapeCombination(a, args) ->
    let traversed = args |> List.map traverse
    ExprShape.RebuildShapeCombination(a, traversed)

let result =
